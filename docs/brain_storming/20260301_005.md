## 🧮 simulation.py（骨組み）

```python
import random
from horse import Horse
from race import simulate_race

YEARS = 1000

def run():
	avg_money_history = []
	avg_stat_history = []

	base_average = 55

	for year in range(YEARS):
		horses = [Horse.generate_ai(base_average) for _ in range(20)]
		prize_total = 0

		for _ in range(12):  # 12ヶ月
			winner = simulate_race(horses)
			prize_total += winner.prize

		avg_stat = sum(h.speed for h in horses)/len(horses)

		avg_money_history.append(prize_total)
		avg_stat_history.append(avg_stat)

		# インフレ制御
		base_average += random.uniform(0,2) - (base_average / 200)

	return avg_money_history, avg_stat_history
```
