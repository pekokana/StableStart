# 🐎 改良版コード（1ファイル完結版）

これを `evolution_test.py` として保存してください。

```python
import random
import statistics
import matplotlib.pyplot as plt

GENERATIONS = 100
HORSES_PER_GEN = 30
RACES_PER_GEN = 30

MUTATION_RATE = 0.05
MUTATION_SCALE = 0.1

CAP = 120.0


class Horse:
	def __init__(self, genome, generation):
		self.genome = genome
		self.generation = generation
		self.money = 0

	@staticmethod
	def random_ai(base_gene, generation, type_name):
		if type_name == "speed":
			genome = {
				"speed": random.gauss(base_gene + 0.1, 0.05),
				"stamina": random.gauss(base_gene - 0.05, 0.05),
				"late": random.gauss(base_gene, 0.05)
			}
		else:  # balance
			genome = {
				"speed": random.gauss(base_gene, 0.05),
				"stamina": random.gauss(base_gene, 0.05),
				"late": random.gauss(base_gene, 0.05)
			}
		return Horse(genome, generation)

	def effective_stat(self, gene):
		raw = gene * 100
		return raw / (1 + raw / CAP)

	def race_score(self):
		speed = self.effective_stat(self.genome["speed"])
		stamina = self.effective_stat(self.genome["stamina"])
		late = self.effective_stat(self.genome["late"])

		penalty = max(0, 80 - stamina) * 0.05
		noise = random.gauss(0, 1)

		return speed + late * 0.2 - penalty + noise


def breed(father, mother, generation):
	genome = {}
	for key in father.genome:
		base = (father.genome[key] + mother.genome[key]) / 2
		mutation = random.gauss(0, 0.03)
		if random.random() < MUTATION_RATE:
			mutation += random.uniform(-MUTATION_SCALE, MUTATION_SCALE)
		genome[key] = max(0.1, base + mutation)
	return Horse(genome, generation)


def simulate_type(type_name):
	base_gene = 0.6
	avg_speed_history = []
	bankruptcy_history = []

	horses = [Horse.random_ai(base_gene, 1, type_name)
			  for _ in range(HORSES_PER_GEN)]

	for gen in range(1, GENERATIONS + 1):

		# レース
		for _ in range(RACES_PER_GEN):
			ranked = sorted(horses, key=lambda h: h.race_score(), reverse=True)
			ranked[0].money += 1000
			ranked[1].money += 400
			ranked[2].money += 200

		# 維持費
		bankrupt = 0
		for h in horses:
			cost = 50 + sum(h.genome.values()) * 20
			h.money -= cost
			if h.money < 0:
				bankrupt += 1

		bankruptcy_rate = bankrupt / len(horses)
		bankruptcy_history.append(bankruptcy_rate)

		avg_speed = statistics.mean(h.genome["speed"] for h in horses)
		avg_speed_history.append(avg_speed)

		# 上位選抜
		horses.sort(key=lambda h: h.money, reverse=True)
		breeders = horses[:10]

		# 次世代
		next_gen = []
		while len(next_gen) < HORSES_PER_GEN:
			f = random.choice(breeders)
			m = random.choice(breeders)
			next_gen.append(breed(f, m, gen + 1))

		horses = next_gen

	return avg_speed_history, bankruptcy_history


# 実行
speed_history, speed_bankrupt = simulate_type("speed")
balance_history, balance_bankrupt = simulate_type("balance")

# 可視化
plt.figure()
plt.plot(speed_history)
plt.plot(balance_history)
plt.title("Average Speed Gene Over Generations")
plt.legend(["Speed Line", "Balance Line"])
plt.show()

plt.figure()
plt.plot(speed_bankrupt)
plt.plot(balance_bankrupt)
plt.title("Bankruptcy Rate")
plt.legend(["Speed Line", "Balance Line"])
plt.show()
```
