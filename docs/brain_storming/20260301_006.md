# 🐎 horse.py

```python
import random
import math

CAP = 120.0

class Horse:
	def __init__(self, genome, generation):
		self.genome = genome
		self.generation = generation
		self.age = 2
		self.money = 0

	@staticmethod
	def random_ai(base_gene, generation):
		genome = {
			"speed": random.gauss(base_gene, 0.1),
			"stamina": random.gauss(base_gene, 0.1),
			"late": random.gauss(base_gene, 0.1)
		}
		return Horse(genome, generation)

	def effective_stat(self, gene_value):
		raw = gene_value * 100
		return raw / (1 + raw / CAP)

	def race_score(self, distance):
		speed = self.effective_stat(self.genome["speed"])
		stamina = self.effective_stat(self.genome["stamina"])
		late = self.effective_stat(self.genome["late"])

		stamina_needed = distance / 20
		stamina_ratio = stamina / stamina_needed

		penalty = 0
		if stamina_ratio < 1:
			penalty = (1 - stamina_ratio) ** 2 * 10

		boost = late * 0.1

		noise = random.gauss(0, 1)

		return speed - penalty + boost + noise
```

# 🧬 genetics.py

```python
import random

MUTATION_RATE = 0.05
MUTATION_SCALE = 0.15

def breed(father, mother, generation):
	child_genome = {}

	for key in father.genome.keys():
		base = (father.genome[key] + mother.genome[key]) / 2
		mutation = random.gauss(0, 0.05)

		if random.random() < MUTATION_RATE:
			mutation += random.uniform(-MUTATION_SCALE, MUTATION_SCALE)

		child_genome[key] = max(0.1, base + mutation)

	from horse import Horse
	return Horse(child_genome, generation)
```

# 🏁 race.py

```python
def simulate_race(horses, distance=2000):
	scores = []

	for horse in horses:
		score = horse.race_score(distance)
		scores.append((horse, score))

	scores.sort(key=lambda x: x[1], reverse=True)

	# 賞金配分
	prizes = [1000, 400, 200]

	for i in range(min(3, len(scores))):
		scores[i][0].money += prizes[i]

	return scores
```

# 🚀 main.py（5世代進化シミュレーション）

```python
import random
from horse import Horse
from race import simulate_race
from genetics import breed
from economy import maintenance_cost
import statistics

GENERATIONS = 5
HORSES_PER_GEN = 20
RACES_PER_GEN = 20

def simulate_generation(base_gene, generation):
	horses = [Horse.random_ai(base_gene, generation) for _ in range(HORSES_PER_GEN)]

	for _ in range(RACES_PER_GEN):
		simulate_race(horses)

	# 維持費差引
	for h in horses:
		h.money -= maintenance_cost(h)

	# 上位5頭選抜
	horses.sort(key=lambda h: h.money, reverse=True)
	breeders = horses[:5]

	avg_gene = statistics.mean([h.genome["speed"] for h in horses])

	print(f"Gen {generation} Avg Speed Gene: {avg_gene:.3f}")

	return breeders, avg_gene


def main():
	base_gene = 0.6
	generation = 1

	breeders, avg_gene = simulate_generation(base_gene, generation)

	while generation < GENERATIONS:
		generation += 1

		next_gen = []

		while len(next_gen) < HORSES_PER_GEN:
			father = random.choice(breeders)
			mother = random.choice(breeders)
			child = breed(father, mother, generation)
			next_gen.append(child)

		# レース実施
		for _ in range(RACES_PER_GEN):
			simulate_race(next_gen)

		for h in next_gen:
			h.money -= maintenance_cost(h)

		next_gen.sort(key=lambda h: h.money, reverse=True)
		breeders = next_gen[:5]

		avg_gene = sum(h.genome["speed"] for h in next_gen) / len(next_gen)

		print(f"Gen {generation} Avg Speed Gene: {avg_gene:.3f}")

if __name__ == "__main__":
	main()
```
